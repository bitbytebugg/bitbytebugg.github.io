////// Java dynamic proxy use pattern
import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;
import java.util.HashMap;
import java.util.Map;

class DynamicInvocationHandler implements InvocationHandler {
    // 被代理的真实对象
    private Object target;

    public DynamicInvocationHandler() {
        this.target = new HashMap<String, String>();
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        String name = method.getName();
        System.out.println("do before method: " + name);
        // 调用真实对象方法
        Object ret = method.invoke(target, args);
        System.out.println("do after method: " + name);
        return ret;
    }
}
public class JavaProxyAnalysis {
    public static void main(String[] args) {
        Map proxyInstance = (Map) Proxy.newProxyInstance(
                HashMap.class.getClassLoader(),
                new Class[]{Map.class},  // 指定哪些接口方法会被代理
                new DynamicInvocationHandler());
        // 非Map接口的方法不会被代理
        proxyInstance.getClass();
        // Map接口的方法会被代理
        proxyInstance.put("key", "value");
    }
}

public static Object newProxyInstance(
        ClassLoader loader,
        Class<?>[] interfaces,
        InvocationHandler h) {
    // 1. 查找代理类
    Class<?> cl = getProxyClass0(loader, intfs);
    // 2. 查找到代理类的构造函数
    final Constructor<?> cons = cl.getConstructor(constructorParams);
    // 3. 调用构造函数生成代理类实例
    return cons.newInstance(new Object[]{h});
}

//////  Proxy.java
private static Class<?> getProxyClass0(ClassLoader loader,
									   Class<?>... interfaces) {
/* Class<?>...类型声明包含了泛型和可变参数的特性：
1. Class是Java类的元数据对象（如String.class、Integer.class）
2. <?> 是泛型通配符，表示可以接受任何类型的Class对象
3. ...表示可变参数，允许传入任意数量的 Class<?>参数，在方法内部会被当作 Class<?>[] 数组处理。
*/
	if (interfaces.length > 65535) {
		throw new IllegalArgumentException("interface limit exceeded");
	}

	// If the proxy class defined by the given loader implementing
	// the given interfaces exists, this will simply return the cached copy;
	// otherwise, it will create the proxy class via the ProxyClassFactory
	return proxyClassCache.get(loader, interfaces);
}

private static final WeakCache<ClassLoader, Class<?>[], Class<?>>
        proxyClassCache = new WeakCache<>(new KeyFactory(), new ProxyClassFactory());

//////  Proxy.java
// ProxyClassFactory::apply()逻辑伪代码实现
public Class<?> ProxyClassFactory::apply(ClassLoader loader, 
		Class<?>[] interfaces) {
	// 1. 对接口列表做校验
    for (intf : interfaces) {
	    // 1.1 验证类加载器 loader 能否解析该接口 intf
	    // 1.2 验证 intf 实际是一个接口
	    // 1.3 验证接口列表中的接口没有重复的，否则直接抛异常
    }
    // 2. 解决代理类和接口列表的权限问题
    for (Class<?> intf : interfaces) {
		int flags = intf.getModifiers();
		// 2.1 如果接口是非public的，需要修改生成代理类的包名，否则跨包无法实现
		if (!Modifier.isPublic(flags)) {
			// 设置代理类的访问修饰符
			accessFlags = Modifier.FINAL;
			String name = intf.getName();
			// 最后一个.的后面是类名
			int n = name.lastIndexOf('.');
			// 获取包名
			String pkg = ((n == -1) ? "" : name.substring(0, n + 1));
			if (proxyPkg == null) {
				proxyPkg = pkg;
			} else if (!pkg.equals(proxyPkg)) {
			// 如果存在多个不同包的私有接口，那么肯定无法同时实现这些接口，抛出异常
				throw new IllegalArgumentException(
					"non-public interfaces from different packages");
			}
		}
	}
	// 3. 如果都是public的接口，就用默认包名 com.sun.proxy
	if (proxyPkg == null) {
		proxyPkg = ReflectUtil.PROXY_PACKAGE + ".";
	}
	// 4. 获取一个静态自增的int，可见运行时生成的代理类最后的数字是一直递增的，不会冲突
	long num = nextUniqueNumber.getAndIncrement();
	String proxyClassNamePrefix = "$Proxy";
	// 5. 得到生成代理类的类名
	String proxyName = proxyPkg + proxyClassNamePrefix + num;

	// 6. 重点来了，生成代理类的字节码
	byte[] proxyClassFile = ProxyGenerator.generateProxyClass(
                proxyName, interfaces, accessFlags);
    // 7. 字节码转换成运行时元类对象
    return defineClass0(loader, proxyName, proxyClassFile, 0, proxyClassFile.length);
}

////// sun/misc/ProxyGenerator.java
public static byte[] generateProxyClass(final String name,
                                        Class<?>[] interfaces,
                                        int accessFlags) {
	ProxyGenerator gen = new ProxyGenerator(name, interfaces, accessFlags);
	// 逻辑交给 generateClassFile()
	final byte[] classFile = gen.generateClassFile();
	// 默认是不生成磁盘上的.class文件。
	if (saveGeneratedFiles) {
		java.security.AccessController.doPrivileged(
		new java.security.PrivilegedAction<Void>() {
			public Void run() {
				try {
					int i = name.lastIndexOf('.');
					Path path;
					if (i > 0) {
						Path dir = Paths.get(name.substring(0, i).replace('.', File.separatorChar));
						Files.createDirectories(dir);
						path = dir.resolve(name.substring(i+1, name.length()) + ".class");
					} else {
						path = Paths.get(name + ".class");
					}
					Files.write(path, classFile);
					return null;
				} catch (IOException e) {
					throw new InternalError(
						"I/O exception saving generated file: " + e);
				}
			}
		});
	}

	return classFile;
}

////// sun/misc/ProxyGenerator.java
private byte[] generateClassFile() {
	/*
	1. Object 方法预处理
	首先无论是什么类，都是继承自Object的，因此Object中的方法是一定需要的。
	注意，这里addProxyMethod并非直接写字节码了，而是做了一些预处理。
	所谓的预处理就是把函数要素：方法名，参数类型，返回类型，异常类型存到一个
	ProxyMethod对象里，并将该对象加入到一个内部维护的列表上
	*/
	addProxyMethod(hashCodeMethod, Object.class);
	addProxyMethod(equalsMethod, Object.class);
	addProxyMethod(toStringMethod, Object.class);
	// 2. 接口列表方法预处理
	for (Class<?> intf : interfaces) {
		for (Method m : intf.getMethods()) {
			addProxyMethod(m, intf);
		}
	}
	for (List<ProxyMethod> sigmethods : proxyMethods.values()) {
		checkReturnTypes(sigmethods);
	}
	// 3. 写入方法和方法对应静态字段的字节码
	// 3.1 写入构造方法字节码到一个MethodInfo实例里
	methods.add(generateConstructor());
	for (List<ProxyMethod> sigmethods : proxyMethods.values()) {
		for (ProxyMethod pm : sigmethods) {
			// 3.2 add static field for method's Method object
			// methodFieldName字段是在ProxyMethod类内部生成的，格式是 m{method_idx}
			// 相当于把Method类型作为生成代理类的静态字段，有多少个代理方法，就有多少个该静态字段。因为 Object::hashCode()是第一个加入的代理方法，所以对应的静态字段为 `Method m1`
			fields.add(new FieldInfo(pm.methodFieldName,
				"Ljava/lang/reflect/Method;",
				 ACC_PRIVATE | ACC_STATIC));
			// 3.3 generate code for proxy method and add it
			methods.add(pm.generateMethod());
		}
	}
	// 3.4 写入<clinit>（Class Initialization Method）的字节码
	methods.add(generateStaticInitializer());
	
	// 4. 写入字节码基本信息到字节数组
	// 5. 调用cp.write(dout)写入常量池信息到字节数组中
	//    cp的内容在前面调用 new FieldInfo() 和 pm.generateMethod()时就已经在构造了。
	// 6. 遍历 fields 调用 field.write(dout) 写入到字节数组中
	// 7. 遍历 methods 调用 method.write(dout) 写入到字节数组中
	// 8. 返回字节数组
	return bout.toByteArray();
}

JNIEXPORT jclass JNICALL
Java_java_lang_reflect_Proxy_defineClass0(JNIEnv *env,
                                          jclass ignore,
                                          jobject loader,
                                          jstring name,
                                          jbyteArray data,
                                          jint offset,
                                          jint length) {
    // 返回值
    jclass result = 0; 
    /* DefineClass 是一个核心的底层原生（Native）函数，负责将字节数组动态转换为可执行
       的Java 类。它是 Java 动态类加载机制的基石之一，尤其在反射、动态代理和字节码增强
       等技术中起关键作用。
    */

    return result;
}
